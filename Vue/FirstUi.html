# @getData

getData(){
    this.loading = true;
    this.more = "loading";
    $METHODS$.then(res=>{
        let {code,data,msg} = res;
        if (code === 200){
            this.list.push(data);
        }else{
            Message.error(msg);
        }
        this.more = "noMore";
        this.isEnd = false;
        this.loading = false;
        // 关闭下拉刷新
        uni.stopPullDownRefresh();
    });
},

# @submitForm

submitLogin() {
    this.$refs.form.validator(this.form, this.rules).then(valid => {
        if (valid.isPassed){
            login(this.form).then(res=>{
                if (res.code === 200){
                    let {access_token,refresh_token,user} = res.data;
                    LocalStorage.set(EnumData.tokenLabel,access_token);
                    LocalStorage.set(EnumData.refreshTokenLabel,refresh_token);
                    this.$store.commit('user/setUser',user);
                    this.fui.toast('登录成功！')
                    router.home.index();
                }else{
                    Message.error(res.msg);
                }
            });
        }
    }).catch(err=>{
        Message.error(err.errorMsg);
        return false;
    })
}
